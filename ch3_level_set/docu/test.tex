\section{example of pseudo code}


    \begin{algorithm}[htbp]
    \SetAlgoLined
    \KwResult{local minimizer $\mathbf{x}^*$}
     $\mathbf{x} = \mathbf{x}_0,\lambda = \lambda_0, num\_step = 0$\\
     \While{$num\_step < max\_num\_step\; \textbf{and} \;$}{
        $\triangle \mathbf{x}$ = get\_update\_step($\mathbf{x}$,$\lambda$)\;
      \eIf{is converge}{
          break;
       }{
         $num\_step = num\_step + 1$\;
         $\mathbf{x}_{\text{new}} = \mathbf{x} + \triangle \mathbf{x}$\; 
         \eIf{cost drop with $\mathbf{x}_{\text{new}}$}{
            $\mathbf{x} = \mathbf{x}_{\text{new}}$\;
            decrease($\lambda$)\;
         }{
            increase($\lambda$)\;
         }
      }
    }
    $\mathbf{x}* = \mathbf{x}$\;
    \caption{The Levenberg-Marquardt method}
\end{algorithm}
